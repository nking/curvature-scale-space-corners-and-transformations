/**
 The quantum register
 */
public class QuantumReg {
    
    /** number of qubits in the qureg */
    public int width;

    /** number of non-zero vectors */    
    public int size;
  
    /** width of the hash array */
    public int hashw;
    
    /** vectors w/ non-zero probability amplitudes.
     * The node's state is a bit vector of superposition of
     * states.
     */
    public QuantumRegNode[] node;
    
    /** hash holding indexes of node.  the hash length
     is usually instantiated as (1 leftshift hashw).
     The hash is used by quantum_get_state to find the node
     index having a given state.
     */
    public int[] hash;

}
