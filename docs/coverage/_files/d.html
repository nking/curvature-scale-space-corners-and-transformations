951     * For each bin:
952     *   (sigma)^2 = (ave sigma_from_all)^2 + (sigma from all points in bin, in function that reduces it by the number of points)
953     *
954     * The total of all bins should be equal to the sigma_from_all;
955     *
956     * N = 3 bins:
957     *    [0] (sigma_bin0)^2 = (sigma_from_all/nBins)^2 + (sigma_bin0 * Fnc(bin0))^2
958     *    [1] (sigma_bin1)^2 = (sigma_from_all/nBins)^2 + (sigma_bin1 * Fnc(bin1))^2
959     *    [2] (sigma_bin2)^2 = (sigma_from_all/nBins)^2 + (sigma_bin2 * Fnc(bin2))^2
960     * Where (sigma_bin0)^2 is the sum of all points that went into bin0 added in quadrature.
961     *
962     * The total of all 3 should equal the total from all points added in quadrature and no more.
963     *
964     *    (sigma_from_all_points)^2 = nBins*(sigma_from_all/nBins)^2 + (sigma_bin0 * Fnc(bin0))^2

967     *    (sigma_from_all_points)^2 * (1 - (1/nBins)) = (sigma_bin0 * Fnc(bin0))^2 + (sigma_bin1 * Fnc(bin1))^2 + (sigma_bin2 * Fnc(bin2))^2
968     *

970     *     Each bin should have a contribution from all points and then a contribution from its own.
971     *     In order for the total to not exceed the sum in quadrature of all points, a bin's sigma squared should
972     *     be the ave from all + it's own times (1-1/N) roughly, which is what the equation above suggests.

1058        float[] xErrors, float[] yErrors) {

1061         *

1064         *
1065         * Errors in histogram:
1066         *
1067         *    error in Y is sqrt(Y) and that is already in standard units.

1090 
1091            float s = sigmaDfDxSquared + sigmaDfDySquared + covXY;
1092 
1093            sum += s;
1094        }
1095 
1096        sum = (float) Math.sqrt(sum);
1097
