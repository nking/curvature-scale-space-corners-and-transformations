401    public long approximateMemoryUsed() {
402 
403        String arch = System.getProperty("sun.arch.data.model");
404 
405        boolean is32Bit = ((arch != null) && arch.equals("64")) ? false : true;
406 
407        int nbits = (is32Bit) ? 32 : 64;
408 
409        int overheadBytes = 16;
410 
411        int intBytes = (is32Bit) ? 4 : 8;
412        int arrayBytes = 32/8;
413        int refBytes = nbits/8;
414 
415        /*
416         * enums:  one has 4 items
417         *         one has 3 items
418         */
419        long sumBits = 4*nbits;
420        long tmpSumBytes = (sumBits/8) + overheadBytes;
421        long padding = (tmpSumBytes % 8);
422        long sumBytes = tmpSumBytes + padding;
423        sumBits = 3*nbits;
424        tmpSumBytes = (sumBits/8) + overheadBytes;
425        padding = (tmpSumBytes % 8);
426        sumBytes += (tmpSumBytes + padding);
427 
428        // a reference to each of the enums
429        sumBytes += (2*intBytes);
430 
431        sumBytes += indexer.approximateMemoryUsed();
432 
433        if (groupHullIndexes != null) {
434            sumBytes += (arrayBytes + (groupHullIndexes.length*(overheadBytes + refBytes + intBytes)));
435        }
436        if (groupFinder != null) {
437            sumBytes += groupFinder.approximateMemoryUsed();
438        }
439        if (backgroundStats != null) {
440            sumBytes += backgroundStats.approximateMemoryUsed();
441        }
442        if (indexerFilePath != null) {
443            // String size on the heap = reference size + content size?
444            sumBytes += (intBytes + (indexerFilePath.length()*intBytes));
445        }
446        if (minimaStatsFilePath != null) {
447            sumBytes += (intBytes + (minimaStatsFilePath.length()*intBytes));
448        }
449 
450        // 8 variables in the stack are each word size
451        sumBytes += 8 * intBytes;
452 
453        // log is reference size on the heap
454        sumBytes += intBytes;
455 
456        sumBytes += overheadBytes;
457 
458        // amount of padding needed to make it a round 8 bytes
459        padding = (sumBytes % 8);
460 
461        sumBytes += padding;
462 
463        return sumBytes;
464    }
