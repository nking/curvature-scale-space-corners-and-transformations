380        }

386        }
387        if (indexerFilePath != null) {

390        }

392            sumBytes += (intBytes + (minimaStatsFilePath.length()*intBytes));
393        }
394 
395        // 8 variables in the stack are each word size
396        sumBytes += 8 * intBytes;
397 
398        // log is reference size on the heap
399        sumBytes += intBytes;
400 
401        sumBytes += overheadBytes;
402 
403        // amount of padding needed to make it a round 8 bytes
404        padding = (sumBytes % 8);
405 
406        sumBytes += padding;
407 
408        return sumBytes;
409    }

551    protected void temporaryWorkaroundForSampling() throws TwoPointVoidStatsException, IOException {
552 
553        // This is a temporary work around to account for not being able to distinguish yet for the default case
554        //   of sampling.  If we used TwoPointVoidStats.Sampling.SEMI_COMPLETE,
555        //   and if there are many points are outliers or all points are in one cluster
556        //   we have to redo-the analysis with TwoPointVoidStats.Sampling.SEMI_COMPLETE_RANGE_SEARCH
557 
558        if (backgroundStats == null) {
559            return;
560        }
561 
562        if (((TwoPointVoidStats)backgroundStats).getSampling().ordinal() == TwoPointVoidStats.Sampling.SEMI_COMPLETE.ordinal()) {
563 
564            // this is very roughly trying to determine whether there are many
565            //   outliers, and if so, we should use the range search because
566            //   it samples between the clusters better.
567            // Caveat is should not reduce the sampling to 'range search' if there are few points already.
568 
569            int nDensities = ((TwoPointVoidStats)backgroundStats).nTwoPointSurfaceDensities;
570 
571if (nDensities < 1001) {
572    return;
573}
574 
575            float fracPoints = calculateFractionOfPointsOutsideOfClusters();
576 
577            float fracArea = calculateFractionOfAreaOutsideOfClusters();
578 
579            int n = (groupFinder != null) ? groupFinder.getNumberOfGroups() : 0;
580 
581            if (fracPoints <= 0.11f) {
582                if (fracArea > 0.33f) {
583                    log.info("KEEP " + TwoPointVoidStats.Sampling.SEMI_COMPLETE.name()
584                        + " fracPoints=" + fracPoints + " fracArea=" + fracArea + " nGroups=" + n);
585                    return;
586                }
587            }
588            log.info("CHANGE TO " + TwoPointVoidStats.Sampling.SEMI_COMPLETE_RANGE_SEARCH.name()
589                + " fracPoints=" + fracPoints + " fracArea=" + fracArea + " nGroups=" + n);
590 
591            if (debug) {
592                plotClusters();
593            }
594 
595            TwoPointVoidStats voidStats = new TwoPointVoidStats(indexer);
596            voidStats.setDebug(debug);
597 
598            if (doLogPerformanceMetrics) {
599                voidStats.logPerformanceMetrics();
600            }
601            
602            if (setUseDownhillSimplexHistogramFitting) {
603                voidStats.setUseDownhillSimplexHistogramFitting();
604            }
605 
606            voidStats.setUseSemiCompleteRangeSampling();
607 
608            voidStats.calc();
609 
610            backgroundStats = voidStats;
611 
612            if (persistTheMinimaStats) {
613                minimaStatsFilePath = voidStats.persistTwoPointBackground();
614            }
615 
616            this.backgroundSurfaceDensity = voidStats.getBackgroundSurfaceDensity();
617            this.backgroundError = voidStats.getBackgroundSurfaceDensityError();
618 
619            if (debug) {
620                log.info("==>background density ="
621                    + this.backgroundSurfaceDensity + " with error =" + this.backgroundError);
622            }
623 
624            findGroups();
625        }

629    /**

633     * @throws TwoPointVoidStatsException
634     */
635    public void calculateHullsOfClusters() throws IOException, TwoPointVoidStatsException {
636 
637        if (state.ordinal() < STATE.CLUSTERS_FOUND.ordinal()) {
638            findClusters();
639        }
640 
641        long startTimeMillis = System.currentTimeMillis();
642        
643        int nGroups = (groupFinder != null) ? groupFinder.getNumberOfGroups() : 0;
644 
645        groupHullIndexes = new SimpleLinkedListNode[nGroups];
646 
647        xGroupHullCentroids = new float[nGroups];
648        yGroupHullCentroids = new float[nGroups];
649        groupHullSurfaceAreas = new float[nGroups];
650 
651        // the indexes stored in the instance vars such as pointToGroupIndex are w.r.t. the arrays sorted by y
652        float[] x = indexer.getXSortedByY();
653        float[] y = indexer.getYSortedByY();
654 
655        int nXY = indexer.getNXY();
656 
657        PolygonAndPointPlotter plotter = new PolygonAndPointPlotter(indexer.getX()[indexer.getSortedXIndexes()[0]],
658            indexer.getX()[indexer.getSortedXIndexes()[nXY - 1]],
659            indexer.getY()[indexer.getSortedYIndexes()[0]], indexer.getY()[indexer.getSortedYIndexes()[nXY - 1]]);
660 
661        for (int i = 0; i < nGroups; i++) {
662 
663            groupHullIndexes[i] = new SimpleLinkedListNode();
664 
665            SimpleLinkedListNode groupNode = groupFinder.getGroupMembershipList()[i];
666 
667            int nMembers = groupNode.getKeys().length;
668 
669            float[] xMember = new float[nMembers];
670            float[] yMember = new float[nMembers];
671            int[] memberIndexes = new int[nMembers];
672 
673            int count = 0;
674 
675            while ((groupNode != null) && (groupNode.key != -1)) {
676 
677                int pointIndex = groupNode.key;
678 
679                xMember[count]       = x[pointIndex];
680                yMember[count]       = y[pointIndex];
681                memberIndexes[count] = pointIndex;
682                count++;
683 
684                groupNode = groupNode.next;
685            }
686 
687            float[] xhull = null;
688            float[] yhull = null;
689 
690            if (count > 2) {
691                try {
692                    GrahamScan scan = new GrahamScan();
693                    scan.computeHull(xMember, yMember);
694 
695                    xhull = scan.getXHull();
696                    yhull = scan.getYHull();
697                } catch (GrahamScanTooFewPointsException e) {
698                }
699            }
700 
701            if (xhull == null) {
702 
703                xhull = new float[count];
704                yhull = new float[count];
705                for (int ii = 0; ii < count; ii++) {
706                    xhull[ii] = xMember[ii];
707                    yhull[ii] = yMember[ii];
708                }
709            }
710 
711            int hullMatchCount = 0;
712            // the hull is a subset of memberIndexes so point to the original coordinates
713            //   instead of creating a new instance data structure
714            for (int j = 0; j < xhull.length; j++) {
715 
716                float xh = xhull[j];
717                float yh = yhull[j];
718 
719                for (int k = 0; k < count; k++) {
720 
721                    int pointIndex = memberIndexes[k];
722 
723                    if ( (x[pointIndex] == xh) && (y[pointIndex] == yh) ) {
724                        groupHullIndexes[i].insert(pointIndex);
725                        hullMatchCount++;
726                        break;
727                    }
728                }
729            }
730 
731            if (debug) {
732                try {
733                    plotter.addPlot(xMember, yMember, xhull, yhull, "");
734                    plotter.writeFile();
735                } catch (IOException e) {
736 
737                }
738            }
739 
740            float[] ca = LinesAndAngles.calcAreaAndCentroidOfSimplePolygon(xhull, yhull);
741 
742            if (ca != null) {
743                groupHullSurfaceAreas[i] = ca[0];
744                xGroupHullCentroids[i] = ca[1];
745                yGroupHullCentroids[i] = ca[2];
746            }
747        }
748 
749        state = STATE.CLUSTER_HULLS_CALCULATED;
750 
751        if (doLogPerformanceMetrics) {
752 
753            long stopTimeMillis = System.currentTimeMillis();
754 
755            printPerformanceMetrics(startTimeMillis, stopTimeMillis, "calculateHullsOfClusters", nGroups);
756        }
757    }
