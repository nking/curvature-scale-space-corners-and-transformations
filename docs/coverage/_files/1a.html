613 
614        long startTimeMillis = System.currentTimeMillis();
615

622        groupHullSurfaceAreas = new float[nGroups];
623 
624        // the indexes stored in the instance vars such as pointToGroupIndex are w.r.t. the arrays sorted by y
625        float[] x = indexer.getXSortedByY();

629

633

635

637

639            float[] xMember = getGroupFinder().getX(i, indexer);

641

662                    yhull[ii] = yMember[ii];

664            }
665

667            // the hull is a subset of memberIndexes so point to the original coordinates
668            //   instead of creating a new instance data structure
669            for (int j = 0; j < xhull.length; j++) {

672                float yh = yhull[j];

691

700                yGroupHullCentroids[i] = ca[2];
701            }
702        }
703

705

707

709 
710            printPerformanceMetrics(startTimeMillis, stopTimeMillis, "calculateHullsOfClusters", nGroups);
