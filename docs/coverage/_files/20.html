383    public long approximateMemoryUsed() {
384 
385        String arch = System.getProperty("sun.arch.data.model");
386 
387        boolean is32Bit = ((arch != null) && arch.equals("64")) ? false : true;
388 
389        int nbits = (is32Bit) ? 32 : 64;
390 
391        int overheadBytes = 16;
392 
393        int intBytes = (is32Bit) ? 4 : 8;
394        int arrayBytes = 32/8;
395        int refBytes = nbits/8;
396 
397        /*
398         * enums:  one has 4 items
399         *         one has 3 items
400         */
401        long sumBits = 4*nbits;
402        long tmpSumBytes = (sumBits/8) + overheadBytes;
403        long padding = (tmpSumBytes % 8);
404        long sumBytes = tmpSumBytes + padding;
405        sumBits = 3*nbits;
406        tmpSumBytes = (sumBits/8) + overheadBytes;
407        padding = (tmpSumBytes % 8);
408        sumBytes += (tmpSumBytes + padding);
409 
410        // a reference to each of the enums
411        sumBytes += (2*intBytes);
412 
413        sumBytes += indexer.approximateMemoryUsed();
414 
415 
416        if (groupMembership != null) {
417            // SimpleLinkedListNode[]
418            sumBytes += (arrayBytes + (groupMembership.length*(overheadBytes + refBytes + intBytes)));
419        }
420        if (groupHullIndexes != null) {
421            sumBytes += (arrayBytes + (groupHullIndexes.length*(overheadBytes + refBytes + intBytes)));
422        }
423        if (xGroupHullCentroids != null) {
424            // float[] on the heap
425            sumBytes += 3*(arrayBytes + (xGroupHullCentroids.length*(arrayBytes)));
426        }
427        if (pointToGroupIndex != null) {
428            sumBytes += (arrayBytes + (pointToGroupIndex.length*arrayBytes));
429        }
430        if (backgroundStats != null) {
431            sumBytes += backgroundStats.approximateMemoryUsed();
432        }
433        if (indexerFilePath != null) {
434            // String size on the heap = reference size + content size?
435            sumBytes += (intBytes + (indexerFilePath.length()*intBytes));
436        }
437        if (minimaStatsFilePath != null) {
438            sumBytes += (intBytes + (minimaStatsFilePath.length()*intBytes));
439        }
440 
441        // 8 variables in the stack are each word size
442        sumBytes += 8 * intBytes;
443 
444        // log is reference size on the heap
445        sumBytes += intBytes;
446 
447        sumBytes += overheadBytes;
448 
449        // amount of padding needed to make it a round 8 bytes
450        padding = (sumBytes % 8);
451 
452        sumBytes += padding;
453 
454        return sumBytes;
455    }

501    public void bruteForceCalculateGroups() {
502 
503        long startTimeMillis = System.currentTimeMillis();
504 
505        // use temporary variables first to prune groups with less than 3 members
506 
507        SimpleLinkedListNode[] tmpGroupMembership = new SimpleLinkedListNode[10];
508        int nTmpGroups = 0;
509        for (int i = 0; i < tmpGroupMembership.length; i++) {
510            tmpGroupMembership[i] = new SimpleLinkedListNode();
511        }
512 
513        int[] tmpPointToGroupIndex = new int[indexer.nXY];
514        Arrays.fill(tmpPointToGroupIndex, -1);
515 
516        // the indexes stored in the instance vars such as pointToGroupIndex are w.r.t. the arrays sorted by y
517        float[] x = indexer.getXSortedByY();
518        float[] y = indexer.getYSortedByY();
519 
520        float densityThreshold = sigmaFactor*backgroundSurfaceDensity;
521        // 2 or 3 times the background density
522        if (debug) {
523            log.info("For clusters, using densityThreshold=" + densityThreshold);
524        }
525 
526        SimpleLinkedListNode tmpPointsInMoreThanOneGroup = new SimpleLinkedListNode();
527 
528        // store the distances between points when the separation implies it is above density
529        for (int i = 0; i < indexer.nXY; i++) {
530 
531            // groups are points whose separation is an implied density higher than 2 or 3 times the background.
532            // see if this point is part of a group already, by looking for it in the pointTo2ndPointIndex entry.
533            // if it is a part of a group already, set the groupNumber,
534            // else, leave the groupNumber as -1 until another point is close enough to start a new group.
535 
536            // this will be -1 if not yet assigned
537            int groupNumber = tmpPointToGroupIndex[i];
538 
539            for (int j = (i + 1); j < indexer.nXY; j++) {
540 
541                float d = (float) Math.sqrt( Math.pow((x[i] - x[j]), 2) + Math.pow((y[i] - y[j]), 2) );
542 
543                float dens = 2.f/d;
544 
545                if (dens > densityThreshold) {
546 
547                    if (tmpGroupMembership.length < (nTmpGroups + 1)) {
548                        int oldN = tmpGroupMembership.length;
549                        int n = (int) (1.5f * oldN);
550                        if (n < (oldN + 1)) {
551                            n = oldN + 1;
552                        }
553                        tmpGroupMembership = Arrays.copyOf(tmpGroupMembership, n);
554                        for (int k = oldN; k < n; k++) {
555                            tmpGroupMembership[k] = new SimpleLinkedListNode();
556                        }
557                    }
558 
559                    if (groupNumber == -1) {
560 
561                        if (tmpPointToGroupIndex[j] != -1) {
562 
563                            groupNumber = tmpPointToGroupIndex[j];
564 
565                        } else {
566 
567                            groupNumber = nTmpGroups;
568                        }
569                        tmpGroupMembership[groupNumber].insert(i);
570                        tmpPointToGroupIndex[i] = groupNumber;
571                        nTmpGroups++;
572 
573                    }// else, is already a member of a group
574 
575                    if ( (tmpPointToGroupIndex[j] != -1) && ( tmpPointToGroupIndex[j] != groupNumber ) ) {
576 
577                        tmpPointsInMoreThanOneGroup.insertIfDoesNotAlreadyExist(j);
578                    }
579 
580                    tmpGroupMembership[groupNumber].insertIfDoesNotAlreadyExist(j);
581 
582                    tmpPointToGroupIndex[j] = groupNumber;
583                }
584            }
585        }
586 
587        // consolidate overlapping groups.  high density backgrounds will have larger number of overlapping clusters
588        SimpleLinkedListNode latest = tmpPointsInMoreThanOneGroup;
589        while (latest != null && latest.key != -1) {
590 
591            int pointIndex = latest.key;
592 
593            int gn1 = tmpPointToGroupIndex[pointIndex];
594 
595            for (int i = 0; i < nTmpGroups; i++) {
596 
597                if (i == gn1) {
598                    continue;
599                }
600 
601                SimpleLinkedListNode grNode = tmpGroupMembership[i].search(pointIndex);
602 
603                if (grNode != null) {
604 
605                    int n1 = tmpGroupMembership[gn1].getNumberOfKeys();
606                    int n2 = tmpGroupMembership[i].getNumberOfKeys();
607 
608                    if (n1 > n2) {
609                        //put all of group 2 points into group 1
610                        SimpleLinkedListNode group2 = tmpGroupMembership[i];
611                        while ( (group2 != null) && (group2.key != -1) ) {
612 
613                            int aPointInG2 = group2.key;
614 
615                            tmpPointToGroupIndex[aPointInG2] = gn1;
616 
617                            tmpGroupMembership[gn1].insertIfDoesNotAlreadyExist(aPointInG2);
618 
619                            tmpGroupMembership[i].delete(aPointInG2);
620 
621                            //group2 = group2.next;
622                            //instead of using next, start from top again due to delete while in iteration
623                            group2 = tmpGroupMembership[i];
624                        }
625                    } else {
626                        //put all of group 1 points into group 2
627                        SimpleLinkedListNode group1 = tmpGroupMembership[gn1];
628                        while ( (group1 != null) && (group1.key != -1) ) {
629 
630                            int aPointInG1 = group1.key;
631 
632                            tmpPointToGroupIndex[aPointInG1] = i;
633 
634                            tmpGroupMembership[i].insertIfDoesNotAlreadyExist(aPointInG1);
635 
636                            tmpGroupMembership[gn1].delete(aPointInG1);
637 
638                            //group2 = group2.next;
639                            //instead of using next, start from top again due to delete while in iteration
640                            group1 = tmpGroupMembership[gn1];
641                        }
642                    }
643 
644                    i = nTmpGroups;
645                }
646            }
647            tmpPointsInMoreThanOneGroup.delete(pointIndex);
648            latest = tmpPointsInMoreThanOneGroup;
649        }
650 
651        // iterate over group data and store in instance vars when group membership is > 2
652        for (int i = 0; i < nTmpGroups; i++) {
653 
654            int currentTmpGroupNumber = i;
655 
656            SimpleLinkedListNode grpMembersNode = tmpGroupMembership[currentTmpGroupNumber];
657 
658            int nGrpMembers = grpMembersNode.getNumberOfKeys();
659 
660            if (nGrpMembers >= minimumNumberInCluster) {
661 
662                // store these associated with current group number which is nGroups
663 
664                int groupNumber = nGroups;
665                nGroups++;
666 
667                while ( (grpMembersNode != null) && (grpMembersNode.key != -1) ) {
668 
669                    int pointIndex = grpMembersNode.key;
670 
671                    checkAndExpandGroupMembershipArray();
672 
673                    pointToGroupIndex[pointIndex] = groupNumber;
674 
675                    groupMembership[groupNumber].insertIfDoesNotAlreadyExist(pointIndex);
676 
677                    grpMembersNode = grpMembersNode.next;
678                }
679            }
680        }
681 
682        state = STATE.CLUSTERS_FOUND;
683 
684        if (doLogPerformanceMetrics) {
685 
686            long stopTimeMillis = System.currentTimeMillis();
687 
688            printPerformanceMetrics(startTimeMillis, stopTimeMillis, "bruteForceCalculateGroups");
689        }
690

692            temporaryWorkaroundForSampling();
693        } catch (IOException e) {
694            log.severe(e.getMessage());
695        } catch (TwoPointVoidStatsException e) {
696            log.severe(e.getMessage());
697        }

823

825        }

850        float[] x = indexer.getXSortedByY();
