417     *
418     *                                | df |^2               | df |^2         df   df

420     *                                | dx |                 | dy |           dx   dy

427     *                 X             dx      X^2
428     *
429     *                                  |   -N   |^2

431     *                                  |   X^2  |
432     *
433     * @param densities - the two point densities
434     * @param point1Indexes indexes to xp, yp of one of the 2 points in the two-point densities
435     * @param point2Indexes indexes to xp, yp of the other of the 2 points in the two-point densities
436     * @param xp x array of points
437     * @param yp y array of points
438     * @param xpe errors in x
439     * @param ype errors in y
440     * @return errors for densities
441     */

443        float[] xp, float[] yp, float[] xpe, float[] ype) {
444 
445        if (densities == null) {
446            throw new IllegalArgumentException("densities cannot be null");
447        }
448        if (point1Indexes == null || point2Indexes == null) {
449            throw new IllegalArgumentException("pointIndexes cannot be null");
450        }
451        if (xp == null || yp == null) {
452            throw new IllegalArgumentException("xp and yp cannot be null");
453        }
454        if (xpe == null || ype == null) {
455            throw new IllegalArgumentException("errors are required");
456        }
457 
458        float[] densityErrors = new float[densities.length];
459 
460        for (int i = 0; i < densities.length; i++) {
461 
462            if (Float.isInfinite(densities[i])) {
463                continue;
464            }
465 
466            int pt1 = point1[i];
467            int pt2 = point2[i];
468 
469            double xe1 = xpe[pt1] ; // xp[pt1];
470            double xe2 = xpe[pt2] ; // xp[pt2];
471            double ye1 = ype[pt1] ; // yp[pt1];
472            double ye2 = ype[pt2] ; // yp[pt2];
473 
474            double xsigmasq = ( xe1 * xe1 ) + ( xe2 * xe2 );
475            double ysigmasq = ( ye1 * ye1 ) + ( ye2 * ye2 );
476            double xysigmasq = xsigmasq + ysigmasq;
477 
478            double linearDensity = allTwoPointSurfaceDensities[i];
479            double linearDensitySq = Math.pow(linearDensity, 2);
480 
481            double err = Math.sqrt(xysigmasq * linearDensitySq);
482 
483            densityErrors[i] = (float)err;
484        }
485        return densityErrors;

710             * Errors in histogram:

713             *    error in X is resolvability, which is bin size = (xHist[1]-xHist[0])/2.
714             *
715             *                                | df |^2               | df |^2         df   df
716             *      (sigma_f)^2 =  (sigma_x)^2|----|   +  (sigma_y)^2|----|    +  2 * -- * -- * cov_ab
717             *                                | dx |                 | dy |           dx   dy
718             *
719             *      For uncorrelated variables the covariance terms are zero.
720             *
721             * Errors in fitting:
722             *
723             *    It looks like the error in the fit could very roughly be math.sqrt(chisq)/nPoints, that is close to
724             *        the chi square statistic which uses degrees of freedom instead of nPoints.
725             *
726             *    But, more formally, errors are based on fitting 2 variable parameters and 1 fixed parameter and on x and y.

740             *      where f_fit = y_const * ----- * exp                           * (1 + k*(-----))
741             *                               sigma                                  (      (sigma))
742             *
743             *      The deriv of f_fit w.r.t. sigma, k, and x is a very very long derivative...
744             *
745             *      If need that formally one day, it's tedious, but possible to work out with the chain rule.
746             *      For now, will instead use chisq to approximate the standard deviation from the model.

763                log.info("estimating background minimum as location of " + limitStr + " of background profile counts = "
764                + this.backgroundSurfaceDensity
765                + " w/ x error in histogram bin =" + errorInEstimateFromHistogram
766                + " gev fitting error for one point =" + errorInFitting);
767            }

770                log.info("WARNING:  chisq is larger than errors: "
771                    + bestFit.getChiSqSum() + " (errsqsum=" + bestFit.getYDataErrSq() + ")");
772            }
