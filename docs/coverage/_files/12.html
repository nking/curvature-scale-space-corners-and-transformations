637

640            }

652                log.info("interpreting the density for a sparse background");

654            } else {
655 
656                //limitStr = "top centroid";

675                    limitError = areaAndXYTopCentroidError;

677                } else {

682                        histogram.getXHist(), histogram.getYHistFloat(), histogram.getXErrors(), histogram.getYErrors(),
683                        0.5f);
684                    
685                    limitError = areaAndXYTopCentroid2Error;
686                }                                
687            }

690 
691            if (debug) {
692                log.info(bestFit.toString());
693            }            
694 
695            this.backgroundDensityError = limitError;
696            
697            if (debug) {
698 
699                // as comparison, log that roughly derived from chi square sum
700                float gevTotalMeanFittingEmpiricalError = (float)Math.sqrt(
701                    GeneralizedExtremeValue.calculateChiSq(bestFit.getX(), bestFit.getYFit()));
702 
703                // empirical error for one bin:

706                String interp = (interpretForSparseBackground) ? " sparse " : " complete ";

708                    + " and interpretation=" + interp

734    protected float[] calcAreaAndCentroid(float[] x, float[] y, int[] regionIndexes) {
735 
736        if (regionIndexes.length == 1) {
737            return null;
738        } else if (regionIndexes.length == 2) {
739 
740            float xp0 = x[regionIndexes[0]];
741            float yp0 = y[regionIndexes[0]];
742 
743            float xp1 = x[regionIndexes[1]];
744            float yp1 = y[regionIndexes[1]];
745 
746            float xc = (xp0 + xp1) / 2.f;
747            float yc = (yp0 + yp1) / 2.f;
748 
749            double dist = Math.sqrt(Math.pow((xp0 - xp1), 2) + Math.pow((yp0 - yp1), 2));
750            double height = (dist > 0) ? 1 : Math.pow(1, Math.getExponent(dist));
751            float area = (float) (height * dist);
752 
753            return new float[]{area, xc, yc};
754        }
755 
756        float[] xPolygon = new float[regionIndexes.length];
757        float[] yPolygon = new float[regionIndexes.length];
758        for (int i = 0; i < regionIndexes.length; i++) {
759            int index = regionIndexes[i];
760            xPolygon[i] = x[index];
761            yPolygon[i] = y[index];
762        }
763 
764        if (xPolygon.length > 2) {
765 
766            try {
767                GrahamScan scan = new GrahamScan();
768                scan.computeHull(xPolygon, yPolygon);
769 
770                float[] xHull = scan.getXHull();
771                float[] yHull = scan.getYHull();
772 
773                return LinesAndAngles.calcAreaAndCentroidOfSimplePolygon(xHull, yHull);
774 
775            } catch (GrahamScanTooFewPointsException e) {
776                return null;
777            }
778 
779        } else {
780 
781            return null;
782        }
783    }
