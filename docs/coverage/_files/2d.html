379        if (zIsNegative) {

381        }

384

386        double df2dsigma = (-1. - (1./k)) * Math.pow(z, (-2. - (1./k)) ) * dzdsigma;
387        
388        //f1 * (1/k) * z^(-1 - (1/k)) * dzdsigma
389        double df1dsigma =  f1 * (1./k) * Math.pow(z, -1. - (1./k)) * dzdsigma;
390        
391        if (zIsNegative) {
392            df2dsigma *= -1.f;
393            df1dsigma *= -1.f;
394        }
395        
396        double df0dsigma = -1.f*yConst/(sigma*sigma);
397                
398        double dydSigma = (  df0dsigma * f1 * f2 ) + ( df1dsigma * f0 * f2 ) + (df2dsigma * f0 * f1 );
399        
400        if (Double.isNaN(dydSigma)) {
401            Double d = estimateDerivUsingDeltaSigma(mu, k, sigma, x);
402            return (d != null) ? d.doubleValue() : 0;
403        }
404        
405        return dydSigma;
406    }

467        }

487        }

524     * at the point right after the model peak.

527     * a fraction of the suggested change.

529     * runtime cost is 
530     *    
531     * @param vars array of current values of k, sigma, and mu
532     * @param varsMin array of minimum allowed values of k, sigma, and mu
533     * @param varsMax array of maximum allowed values of k, sigma, and mu

538     * @param y normalized y values of histogram to fit
539     * @param ye normalized y error values of histogram to fit

541     *     the items are ordered so that index=0 holds delta k, index=1 holds delta sigma,
542     *     index=2 holds delta mu.  the values may be zero if no step was found to

551    public static void derivsThatMinimizeChiSqSum(float[] vars, float[] varsMin, float[] varsMax, float[] chiSqSumReturn,

557        //   we look at the first derivatives of the point in the model GEV right after the maximum
