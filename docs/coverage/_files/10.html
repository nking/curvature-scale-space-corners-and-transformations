440    public long approximateMemoryUsed() {
441 
442        String arch = System.getProperty("sun.arch.data.model");
443 
444        boolean is32Bit = ((arch != null) && arch.equals("64")) ? false : true;
445 
446        int nbits = (is32Bit) ? 32 : 64;
447 
448        int overheadBytes = 16;
449 
450        int intBytes = (is32Bit) ? 4 : 8;
451        int arrayBytes = 32/8;
452        int refBytes = nbits/8;
453 
454        /*
455         * enums:  one has 4 items
456         *         one has 3 items
457         */
458        long sumBits = 4*nbits;
459        long tmpSumBytes = (sumBits/8) + overheadBytes;
460        long padding = (tmpSumBytes % 8);
461        long sumBytes = tmpSumBytes + padding;
462        sumBits = 3*nbits;
463        tmpSumBytes = (sumBits/8) + overheadBytes;
464        padding = (tmpSumBytes % 8);
465        sumBytes += (tmpSumBytes + padding);
466 
467        // a reference to each of the enums
468        sumBytes += (2*intBytes);
469 
470        sumBytes += indexer.approximateMemoryUsed();
471 
472        if (groupFinder != null) {
473            sumBytes += groupFinder.approximateMemoryUsed();
474        }
475        if (backgroundStats != null) {
476            sumBytes += backgroundStats.approximateMemoryUsed();
477        }
478        if (indexerFilePath != null) {
479            // String size on the heap = reference size + content size?
480            sumBytes += (intBytes + (indexerFilePath.length()*intBytes));
481        }
482        if (minimaStatsFilePath != null) {
483            sumBytes += (intBytes + (minimaStatsFilePath.length()*intBytes));
484        }
485 
486        // 8 variables in the stack are each word size
487        sumBytes += 8 * intBytes;
488 
489        // log is reference size on the heap
490        sumBytes += intBytes;
491 
492        sumBytes += overheadBytes;
493 
494        // amount of padding needed to make it a round 8 bytes
495        padding = (sumBytes % 8);
496 
497        sumBytes += padding;
498 
499        return sumBytes;
500    }
