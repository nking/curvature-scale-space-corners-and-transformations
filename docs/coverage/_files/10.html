418        for (int i = 0; i < yFit.getOriginalScaleX().length; i++) {
419            if (i > yPeakIdx) {
420                if (yFit.getOriginalScaleYFit()[i] > yLimit) {
421                    yLimitIdx = i;
422                } else {
423                    break;
424                }
425            } else {
426                yLimitIdx = i;
427            }
428        }
429 
430        // xError[i] should be formally calculated, but the approximation that it can be determined no
431        //   better than  the bin center +- binwidth/2  is a minimum error.
432        //   a safe addition to that (added in quadrature) would be an error in x derived from chi square
433        //   but that is not done here
434 
435        float xDelta = yFit.getX()[1] - yFit.getX()[0];
436        float xErrorSq = (xDelta*xDelta/4.f);
437 
438        float sum = 0.0f;
439        for (int i = 0; i <= yLimitIdx; i++) {
440            sum += xErrorSq;
441        }
442 
443        sum = (float) Math.sqrt(sum);
444 
445        return sum;
446    }

466    public static float[] genCurve(float[] x1, float k, float sigma, float mu) {
467 
468        if (sigma == 0) {
469            return null;
470        }
471        // if k =~ 0, use Gumbel which is Type I
472        if (k == 0 || Float.isInfinite(1.f/k)) {
473            return generateEVTypeICurve(x1, sigma, mu);
474        }
475 
476        float[] yGEV = new float[x1.length];
477 
478        for (int i = 0; i < x1.length; i++) {
479 
480            float z = 1.f + k * ((x1[i] - mu)/sigma);
481            float a,b;
482 
483            boolean zIsNegative = (z < 0);
484            // When z is negative, need to use alternative methods for exponentiation:
485            //
486            // For z^(g/h)
487            //
488            // For the continuous real exponentiation operator, negative base isn't allowed
489            // For the discrete real exponentiation operator,
490            //    fractional exponents with odd denominators are allowed.
491            //    (-z)^(g/h) = ((-z)^g)^(1/h) = ((-1)^g)(z^(g/h))
492            // For the complex exponentiation operator, that is complex bases,
493            //    the results are not continuous and are infinite.
494            //    the principal value is
495            //        (-1)^(g/h) * ((1./z)^(g/h))
496 
497            if (zIsNegative) {
498                float invNegZ = -1.0f*(1.0f/z);
499                float neg1Pow = -1.0f; // TODO:  revisit this
500                a = -1.f * neg1Pow * (float) Math.pow(invNegZ, (-1.f/k));
501                b = neg1Pow * (float) Math.pow(invNegZ, (-1.f - (1.f/k)));
502            } else {
503                a = -1.f * (float) Math.pow(z, (-1.f/k));
504                b = (float) Math.pow(z, (-1.f - (1.f/k)));
505            }
506 
507            if (Float.isNaN(a) || Float.isNaN(b)) {
508                yGEV[i] = 0;
509            } else {
510                float t = (float) ((1.f/sigma) * Math.exp(a) * b);
511                if (t < 0 || Float.isNaN(t)) {
512                    yGEV[i] = 0;
513                } else {
514                    yGEV[i] = t;
515                }
516            }
517        }
518 
519        return yGEV;
520    }
