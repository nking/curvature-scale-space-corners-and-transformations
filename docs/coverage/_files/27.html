413                        ktmp, stmp, mtmp, weightMethod, yNorm);
414                    yfits[midSimplexIndex].setYDataErrSq(yErrSquareSum);
415 
416                } else {
417

419                }
420 
421            } else if (
422                (kReflect >= kMin) && (kReflect <= kMax) 
423                && (sigmaReflect >= sigmaMin) && (sigmaReflect <= sigmaMax)
424                && (muReflect >= muMin) && (muReflect <= muMax)
425                )
426            {
427 
428                yfits[maxSimplexIndex] = yfitReflected;
429            }
430 
431            nIter++;
432 
433            //if (yfits[minSimplexIndex].getChiSqSum() < yErrSquareSum) {

435            //}

439            }
440        }
441 
442        yfits[0].kSolutionResolution = yfits[0].getK() - yfits[1].getK();
443        yfits[0].sigmaSolutionResolution = yfits[0].getSigma() - yfits[1].getSigma();

445        if (yfits[0].getYFit() != null) {

451    }

453    /**
454     * sort the array yfits by ascending chisquare sum using
455     * the quick sort algorithm.
456     *
457     * @param yfits
458     * @param p first index of the array yfits to be sorted
459     * @param r the last index of the array yfits to be sorted, inclusive
460     */
461    void sortFromMinToMax(GEVYFit[] yfits, int p, int r) {
462 
463        if (p < r) {
464 
465            int q = partition(yfits, p, r);
466 
467            sortFromMinToMax(yfits, p, q - 1);
468 
469            sortFromMinToMax(yfits, q + 1, r);
470        }
471    }
472    /**
473     * the partition function of the GEVYFit quick sort method.
474     *
475     * @param yfits
476     * @param p
477     * @param r
478     * @return
479     */
480    int partition(GEVYFit[] yfits, int p, int r) {
481 
482        float xxp = yfits[r].getChiSqSum();
483 
484        int i = p - 1;
485 
486        for (int j = p; j < r ; j++ ) {
487            if (yfits[j].getChiSqSum() <= xxp) {
488 
489                i++;
490 
491                GEVYFit swap = yfits[i];
492                yfits[i] = yfits[j];
493                yfits[j] = swap;
494            }
495        }
496 
497        GEVYFit swap = yfits[i + 1];
498        yfits[i + 1] = yfits[r];
499        yfits[r] = swap;
500 
501        return i + 1;
502    }
