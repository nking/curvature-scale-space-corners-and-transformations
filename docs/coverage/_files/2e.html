404                if (countsBelowMinAtTail > 0) {
405                    maxx = xHist[lastLowCountIdx];
406                    binWidth = calculateBinWidth(minx, maxx, nBins);
407                    Histogram.createHistogram(values, nBins, minx, maxx, xHist, yHist, binWidth);
408                }
409            }
410        }
411        
412        if (values.length > 100) {
413            // if there are a large number of points, we'd like to increase the resolution of the peak if needed
414            int nLeftOfPeak = MiscMath.findYMaxIndex(yHist);
415            int nIter = 0;
416            while (nIter < 30 && nLeftOfPeak < 3 && (yHist[nLeftOfPeak] > 100)) {
417                binWidth *= 0.8f;
418                Histogram.createHistogram(values, nBins, minx, maxx, xHist, yHist, binWidth);
419                nLeftOfPeak = MiscMath.findYMaxIndex(yHist);
420                nIter++;
421            }
422        }
423          
424        float[] yHistFloat = new float[yHist.length];
425        for (int i = 0; i < yHist.length; i++) {
426            yHistFloat[i] = (float) yHist[i];
427        }
428 
429        float[] yErrors = new float[xHist.length];
430        float[] xErrors = new float[xHist.length];
431 
432        calulateHistogramBinErrors(xHist, yHist, values, valueErrors, xErrors, yErrors);
433 
434        HistogramHolder histogram = new HistogramHolder();
435        histogram.setXHist(xHist);
436        histogram.setYHist(yHist);
437        histogram.setYHistFloat(yHistFloat);
438        histogram.setYErrors(yErrors);
439        histogram.setXErrors(xErrors);
440        
441        return histogram;
442    }

446     * to only that peak, else leave unaltered.

448     * @param hist
449     * @param values
450     * @param valueErrors

452     */
453    public static HistogramHolder reduceHistogramToFirstPeak(HistogramHolder 
454        hist, float[] values, float[] valueErrors) {
455        
456        int yPeakIdx = findFirstPeakIndex(hist);
457        
458        if (yPeakIdx == -1) {
459            return hist;
460        }

473        tmp.setXErrors(Arrays.copyOfRange(hist.getXErrors(), 0, n));
474        tmp.setYErrors(Arrays.copyOfRange(hist.getYErrors(), 0, n));

480        
481        float yPeak = Float.MIN_VALUE;
482        int yPeakIdx = -1;
483        
484        // specific to use here, find max within first half of histogram
485        for (int i = 0; i < hist.getXHist().length/2; i++) {
486

488

490                yPeak = y;

492            }
493        }

523        return yPeakMinIdx;
