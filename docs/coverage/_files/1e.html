651    protected void finalizeStats(HistogramHolder histogram, GEVYFit yfit) throws TwoPointVoidStatsException {
652 
653        if (yfit == null) {

655            throw new TwoPointVoidStatsException("yfit cannot be null");
656        }
657        this.statsHistogram = histogram;
658 
659        this.bestFit = yfit;
660 
661        try {
662 
663            if (debug) {
664                log.info(bestFit.toString());
665 
666                float xHalfInterval = (histogram.getXHist()[1] - histogram.getXHist()[0]) / 2.0f;
667                float xmin = 0;
668                float xmax = histogram.getXHist()[histogram.getXHist().length - 1] + xHalfInterval;
669                float ymin = 0;
670                float ymax = MiscMath.findMax(histogram.getYHistFloat());
671 
672                float[] xf = bestFit.getOriginalScaleX();
673                float[] yf = bestFit.getOriginalScaleYFit();
674                PolygonAndPointPlotter plotter = new PolygonAndPointPlotter(xmin, xmax, ymin, ymax);
675                plotter.addPlot(histogram.getXHist(), histogram.getYHistFloat(),
676                    histogram.getXErrors(), histogram.getYErrors(), xf, yf, "");
677                plotter.writeFile3();
678            }
679 
680            /* for interpretForSparseBackground:
681                   background density is the lowest bin's x value in a well formed histogram.
682               for all other sampling:
683                   background density is the peak bin's x value in a well formed histogram.
684            */
685 
686            float limit, limitError;
687 
688            if (interpretForSparseBackground == null) {
689                interpretForSparseBackground = Boolean.FALSE;
690            }
691 
692            if (interpretForSparseBackground) {
693 
694                int yPeakIndex = MiscMath.findYMaxIndex(histogram.getYHist());
695 
696                limit = histogram.getXHist()[0];
697 
698            } else {
699 
700                //limitStr = "top centroid";
701 
702                // centroid of area defined by the top portion of the fit where y >= ypeak/2
703                float[] areaAndXYTopCentroid = calculateCentroidOfTop(bestFit.getOriginalScaleX(), bestFit.getOriginalScaleYFit(), 0.5f);
704 
705                float[] areaAndXYTopCentroid2 = calculateCentroidOfTop(histogram.getXHist(), histogram.getYHistFloat(), 0.5f);
706 
707                float xpeak = (areaAndXYTopCentroid[0] > 0) ? areaAndXYTopCentroid[1] : areaAndXYTopCentroid2[1];
708                
709System.out.println("GEV top centroid=" + areaAndXYTopCentroid[1] + " data top centroid=" + areaAndXYTopCentroid2[1] + " using xpeak=" + xpeak);
710 
711                limit = (xpeak > 0) ? xpeak : bestFit.getXPeak();
712            }
713 
714            this.backgroundSurfaceDensity = limit;
715 
716            if (debug) {
717                log.info(bestFit.toString());
718            }
719 
720            int limitIndex = bestFit.getXPeakIndex();
721            limitError = histogram.getXErrors()[limitIndex];
722 
723 
724            /* Errors add in quadrature:
725             *
726             *    * errors in histogram
727             *    * errors in GEV fitting
728             *
729             * Errors in histogram:
730             *
731             *    error in Y is sqrt(Y) and that is already in standard units.
732             *    error in X is resolvability, which is bin size = (xHist[1]-xHist[0])/2.
733             *
734             *                                | df |^2               | df |^2         df   df
735             *      (sigma_f)^2 =  (sigma_x)^2|----|   +  (sigma_y)^2|----|    +  2 * -- * -- * cov_ab
736             *                                | dx |                 | dy |           dx   dy
737             *
738             *      For uncorrelated variables the covariance terms are zero.
739             *
740             * Errors in fitting:
741             *
742             *    It looks like the error in the fit could very roughly be math.sqrt(chisq)/nPoints, that is close to
743             *        the chi square statistic which uses degrees of freedom instead of nPoints.
744             *
745             *    But, more formally, errors are based on fitting 2 variable parameters and 1 fixed parameter and on x and y.
746             *    The 2 parameters are GEV.sigma and GEV.k (we use a fixed mu usually)
747             *
748             *                                    | df_fit |^2               | df_fit |^2
749             *      (sigma_f_fit)^2 =  (sigma_x)^2|--------|   +  (sigma_y)^2|--------|
750             *                                    |   dx   |                 |   dy   |
751             *
752             *                             | df_fit |^2                       |  df_fit  |^2
753             *          +  (sigma_GEV.k)^2 |--------|   +  (sigma_GEV.sigma)^2|----------|
754             *                             | dGEV.k |                         |dGEV.sigma|
755             *
756             *                                          (   (      ( x-mu))-(1/k))
757             *                                          (-1*(1 + k*(-----))      )
758             *                                 1        (   (      (sigma))      )  (      ( x-mu))(-1-(1/k))
759             *      where f_fit = y_const * ----- * exp                           * (1 + k*(-----))
760             *                               sigma                                  (      (sigma))
761             *
762             *      The deriv of f_fit w.r.t. sigma, k, and x is a very very long derivative...
763             *
764             *      If need that formally one day, it's tedious, but possible to work out with the chain rule.
765             *      For now, will instead use chisq to approximate the standard deviation from the model.
766             */
767 
768            // Error from one value is dependent upon having counted all in histogram.  we take the
769            //  error for our single bin as an average over all because of that.
770            float errorInEstimateFromHistogram = limitError;
771 
772            float gevTotalMeanFittingError =
773                GeneralizedExtremeValue.calculateTotalMeanFittingError(bestFit.getX(), bestFit.getYFit());
774 
775            float errorInFitting = gevTotalMeanFittingError / histogram.getYHist().length;
776 
777            // add errors in quadrature
778            this.backgroundSurfaceDensityError = (float) (Math.sqrt(
779                errorInEstimateFromHistogram * errorInEstimateFromHistogram + errorInFitting * errorInFitting));
780 
781            if (debug) {
782                String interp = (interpretForSparseBackground) ? "sparse" : "complete";
783                log.info("estimating background from sampling= " + sampling.toString()
784                + " and interpretation=" + interp
785                + this.backgroundSurfaceDensity
786                + " w/ x error in histogram bin =" + errorInEstimateFromHistogram
787                + " gev fitting error for one point =" + errorInFitting
788                + " error in the histogram x bin at y peak = " + limitError);
789            }
790 
791            if (debug && (bestFit.getChiSqSum() > bestFit.getYDataErrSq())) {
792                log.info("WARNING:  chisq is larger than errors: "
793                    + bestFit.getChiSqSum() + " (errsqsum=" + bestFit.getYDataErrSq() + ")");
794            }
795 
796            state = State.STATS_FINALIZED;
797 
798        } catch (IOException e) {
799            throw new TwoPointVoidStatsException(e);
800        }
