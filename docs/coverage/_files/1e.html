419    /**
420     * Calculate the 2 point density errors following the chain rule
421     *
422     *                                | df |^2               | df |^2         df   df
423     *      (sigma_f)^2 =  (sigma_x)^2|----|   +  (sigma_y)^2|----|    +  2 * -- * -- * cov_ab
424     *                                | dx |                 | dy |           dx   dy
425     *
426     *      For uncorrelated variables the covariance terms are zero.
427     *
428     * For two-point density:
429     *                 N             dF      -N
430     *      F(x,y) = ------   ==>   ---- =  -----
431     *                 X             dx      X^2
432     *
433     *                                  |   -N   |^2
434     *     (sigma_f)^2 =  (sigma_x)^2 * |--------|
435     *                                  |   X^2  |
436     *
437     * @param densities - the two point densities
438     * @param point1Indexes indexes to xp, yp of one of the 2 points in the two-point densities
439     * @param point2Indexes indexes to xp, yp of the other of the 2 points in the two-point densities
440     * @param xp x array of points
441     * @param yp y array of points
442     * @param xpe errors in x
443     * @param ype errors in y
444     * @return errors for densities
445     */
446    float[] calulateTwoPointDensityErrors(float[] densities, int[] point1Indexes, int[] point2Indexes,
447        float[] xp, float[] yp, float[] xpe, float[] ype) {
448 
449        if (densities == null) {
450            throw new IllegalArgumentException("densities cannot be null");
451        }
452        if (point1Indexes == null || point2Indexes == null) {
453            throw new IllegalArgumentException("pointIndexes cannot be null");
454        }
455        if (xp == null || yp == null) {
456            throw new IllegalArgumentException("xp and yp cannot be null");
457        }
458        if (xpe == null || ype == null) {
459            throw new IllegalArgumentException("errors are required");
460        }
461 
462        float[] densityErrors = new float[densities.length];
463 
464        for (int i = 0; i < densities.length; i++) {
465 
466            if (Float.isInfinite(densities[i])) {
467                continue;
468            }
469 
470            int pt1 = point1[i];
471            int pt2 = point2[i];
472 
473            double xe1 = xpe[pt1] ; // xp[pt1];
474            double xe2 = xpe[pt2] ; // xp[pt2];
475            double ye1 = ype[pt1] ; // yp[pt1];
476            double ye2 = ype[pt2] ; // yp[pt2];
477 
478            double xsigmasq = ( xe1 * xe1 ) + ( xe2 * xe2 );
479            double ysigmasq = ( ye1 * ye1 ) + ( ye2 * ye2 );
480            double xysigmasq = xsigmasq + ysigmasq;
481 
482            double linearDensity = allTwoPointSurfaceDensities[i];
483            double linearDensitySq = Math.pow(linearDensity, 2);
484 
485            double err = Math.sqrt(xysigmasq * linearDensitySq);
486 
487            densityErrors[i] = (float)err;
488        }
489        return densityErrors;
490    }

733             *    * errors in GEV fitting
734             *
735             * Errors in histogram:
736             *
737             *    error in Y is sqrt(Y) and that is already in standard units.
738             *    error in X is resolvability, which is bin size = (xHist[1]-xHist[0])/2.
739             *
740             *                                | df |^2               | df |^2         df   df
741             *      (sigma_f)^2 =  (sigma_x)^2|----|   +  (sigma_y)^2|----|    +  2 * -- * -- * cov_ab
742             *                                | dx |                 | dy |           dx   dy
743             *
744             *      For uncorrelated variables the covariance terms are zero.
745             *
746             * Errors in fitting:
747             *
748             *    It looks like the error in the fit could very roughly be math.sqrt(chisq)/nPoints, that is close to
749             *        the chi square statistic which uses degrees of freedom instead of nPoints.
750             *
751             *    But, more formally, errors are based on fitting 2 variable parameters and 1 fixed parameter and on x and y.
752             *    The 2 parameters are GEV.sigma and GEV.k (we use a fixed mu usually)
753             *
754             *                                    | df_fit |^2               | df_fit |^2
755             *      (sigma_f_fit)^2 =  (sigma_x)^2|--------|   +  (sigma_y)^2|--------|

757             *
758             *                             | df_fit |^2                       |  df_fit  |^2
759             *          +  (sigma_GEV.k)^2 |--------|   +  (sigma_GEV.sigma)^2|----------|
760             *                             | dGEV.k |                         |dGEV.sigma|
761             *
762             *                                          (   (      ( x-mu))-(1/k))
763             *                                          (-1*(1 + k*(-----))      )
764             *                                 1        (   (      (sigma))      )  (      ( x-mu))(-1-(1/k))
765             *      where f_fit = y_const * ----- * exp                           * (1 + k*(-----))
766             *                               sigma                                  (      (sigma))
767             *
768             *      The deriv of f_fit w.r.t. sigma, k, and x is a very very long derivative...
769             *
770             *      If need that formally one day, it's tedious, but possible to work out with the chain rule.
771             *      For now, will instead use chisq to approximate the standard deviation from the model.
772             */
773 
774            // Error from one value is dependent upon having counted all in histogram.  we take the
775            //  error for our single bin as an average over all because of that.
776            float errorInEstimateFromHistogram = limitError;

780

782

784            this.backgroundSurfaceDensityError = (float) (Math.sqrt(
785                errorInEstimateFromHistogram * errorInEstimateFromHistogram + errorInFitting * errorInFitting));
786 
787            if (debug) {
788                log.info("estimating background minimum as location of " + limitStr + " of background profile counts = "
789                + this.backgroundSurfaceDensity
790                + " w/ x error in histogram bin =" + errorInEstimateFromHistogram
791                + " gev fitting error for one point =" + errorInFitting);
792            }
793 
794            if (debug && (bestFit.getChiSqSum() > bestFit.getYDataErrSq())) {
795                log.info("WARNING:  chisq is larger than errors: "
796                    + bestFit.getChiSqSum() + " (errsqsum=" + bestFit.getYDataErrSq() + ")");
797            }
