522    public static double distToLine(float x0, float y0, float x1, float y1,
523        float xPoint, float yPoint) {
524 
525        float xLineDiff = (x1 - x0);
526        float xPointDiff = (xPoint - x0);
527        float yLineDiff = (y1 - y0);
528        float yPointDiff = (yPoint - y0);
529 
530        double prDist = Math.sqrt(  xPointDiff*xPointDiff + yPointDiff*yPointDiff );
531 
532        /*
533         *     (xPoint, yPoint)
534         *        *
535         *       /
536         *   PR /                        perpendicular distance = PR sin theta
537         *     /theta
538         *    *-------------------*
539         *  (x0,y0)            (x1,y1)
540         *
541         *  |A||B|cos(theta) = A.B = (x1-x0)(xPoint-x0) + (y1-y0)(yPoint-y0)
542         *  |A|= sqrt( (x1-x0)^2 + (y1-y0)^2)
543         *  |B|= sqrt( (xPoint-x0)^2 + (yPoint-y0)^2)
544         *
545         *   cos(theta) = (x1-x0)(xPoint-x0) + (y1-y0)(yPoint-y0) / (sqrt( (x1-x0)^2 + (y1-y0)^2) * sqrt( (xPoint-x0)^2 + (yPoint-y0)^2))
546         */
547 
548        double a = (xLineDiff * xPointDiff + yLineDiff * yPointDiff);
549        double b = Math.sqrt(xLineDiff*xLineDiff + yLineDiff*yLineDiff)
550            * Math.sqrt(xPointDiff*xPointDiff + yPointDiff*yPointDiff);
551        double theta = Math.acos( a/b );
552 
553        return prDist*Math.sin(theta);
554    }
555 
556    /*
557     *
558     * determine angle subtended by line and point where we are returning the angle
559     * (NOTE:  set the first point as the one near the ngle of interest)
560     *
561     *      (x1,y1)
562     *         /
563     *        /      *(xPoint,yPoint)
564     *       /
565     *      /theta
566     *   (x0,y0)
567     *
568     *
569     */
570    public static double angleBetweenPointAndLine(float x0, float y0, float x1, float y1,
571        float xPoint, float yPoint) {
572 
573        /*
574         * Use Law of Cosines
575         *
576         *                   (x1, y1)
577         *                      *
578         *                    .
579         *        side c    .  \/.
580         *                .    B               c^2 = a^2 + b^2 - 2ab cos C
581         *              .         .            or  a^2 = b^2 + c^2 - 2bc cos A
582         *            .
583         *          .              . side a
584         * (x0,y0)*  _\A
585         *            .             .
586         *                .
587         *                    .      .
588         *            side b      . C/
589         *                             *(xp,yp)
590         *
591         *  cosine A = (b^2 + c^2 - a^2)/2bc
592         */
593 
594        double bsq = distSquared(x0, y0, xPoint, yPoint);
595        double csq = distSquared(x0, y0, x1, y1);
596        double asq = distSquared(xPoint, yPoint, x1, y1);
597 
598        double cosA = ( bsq + csq - asq)/( 2.*Math.sqrt(bsq) * Math.sqrt(csq));
599        double theta = Math.acos( cosA );// value is in range the 0.0 through pi
600 
601        // use a convention for the sign
602        theta = ((xPoint - x0) > 0) ? theta : -1.*theta;
603 
604        return theta;
605    }
606 
607    /**
608     * For Clockwise, solving theta = C in diagram below.
609     * For Counter Clockwise, solving theta = A in diagram below.
610     *
611     *                   (xp, yp)
612     *                      *
613     *                    .
614     *        side c    .  \/.             Law of cosines
615     *                .    B               c^2 = a^2 + b^2 - 2ab cos C
616     *              .         .            or  a^2 = b^2 + c^2 - 2bc cos A
617     *            .
618     *          .              . side a
619     * (x0,y0)*  _\A
620     *            .             .
621     *                .
622     *                    .      .
623     *            side b      . C/
624     *                             *(x1,y1)
625     *
626     *   Right triangle cosine and sine
627     *     r /
628     *      /
629     *     /theta    xTopPolygon = r * cos(theta)
630     *    -------
631     *       xTopPolygon
632     *
633     * @param x0
634     * @param y0
635     * @param x1
636     * @param y1
637     * @param xPoint
638     * @param yPoint
639     * @param calculateClockWise
640     * @return the angle between the edge 0:1 and edge 1:Point which is C for Clockwise, else A for Counter Clockwise
641     */
642    public static double angleBetweenPointAndLine(float x0, float y0, float x1, float y1,
643        float xPoint, float yPoint, boolean calculateClockWise) {
644 
645        if (x1 < x0) {
646            throw new IllegalArgumentException("Expecting x0 <= x1.");
647        }
648 
649        float slope01 = (y1 - y0)/(x1 - x0);
650        float slope1p = (yPoint - y1)/(xPoint - x1);
651 
652        // FIRST handling the cases where edge 0:1 and/or 1:point are horizontal or vertical
653        //   to use right triangle forumulas
654 
655        double angle = angleBetweenPointAndLineWithRightTriangle(x0, y0, x1, y1, xPoint, yPoint, calculateClockWise);
656 
657        if (angle > -1) {
658                return angle;
659        }
660 
661        // ELSE transform points to reference frame where edge 0:1 is horizontal
662        // And calculate angles using Law Of Cosines.
663        // Then see if 'CW' or 'CCW' w.r.t 0:1 means an inside or outside angle should be
664        //    returned.
665 
666        // Transform points 1 and P so that 0:1 is horizontal.  Using the origin as center of rotation.
667 
668        double thetaTransformCW = 0.;
669       /*   Right triangle cosine and sine
670        *     r /
671        *      /
672        *     /theta    xTopPolygon = r * cos(theta)
673        *    -------
674        *       xTopPolygon
675        */
676        if (x0 == x1) {
677                thetaTransformCW = -90.;
678        } else if (y0 == y1) {
679                thetaTransformCW = 0.;
680        } else {
681                thetaTransformCW = -1*Math.atan2((y1-y0), (x1-x0));
682        }
683 
684        // P'_x = origin_x + ((xTopPolygon-origin_x)*math.cos(theta) - (yTopPolygon-origin_y)*math.sin(theta))
685        // P'_y = origin_y + ((yTopPolygon-origin_y)*math.cos(theta) + (xTopPolygon-origin_x)*math.sin(theta))
686        float x00 = x0;
687        float y00 = y0;
688 
689        float x11 = x0 + (float)((x1 - x0)*Math.cos(thetaTransformCW) - (y1 - y0)*Math.sin(thetaTransformCW));
690        float y11 = y0 + (float)((y1 - y0)*Math.cos(thetaTransformCW) + (x1 - x0)*Math.sin(thetaTransformCW));
691 
692        float xpp = x0 + (float)((xPoint - x0)*Math.cos(thetaTransformCW) - (yPoint - y0)*Math.sin(thetaTransformCW));
693        float ypp = y0 + (float)((yPoint - y0)*Math.cos(thetaTransformCW) + (xPoint - x0)*Math.sin(thetaTransformCW));
694 
695        /*
696         *   right triangles have already been handled, so we can limit cases by (xpp, ypp) position
697         *      relative to 0:1
698         * _________________   _________________   _________________
699         * |case 0:         |  |case 1:         |  |case 2:         |
700         * |                |  |                |  |                |
701         * | P              |  |       P        |  |             P  |   <=== inner angles in polygon
702         * |    *     *     |  |    *     *     |  |    *     *     |
703         * |    0     1     |  |    0     1     |  |    0     1     |
704         * |                |  |                |  |                |
705         * ------------------  ------------------  ------------------
706         *
707         * _________________   _________________   _________________
708         * |case 3:         |  |case 4:         |  |case 5:         |
709         * |                |  |                |  |                |
710         * |                |  |                |  |                |  <=== the inner angle is measured
711         * |    *     *     |  |    *     *     |  |    *     *     |       but we want the outer angle
712         * |    0     1     |  |    0     1     |  |    0     1     |       for the answer
713         * | P              |  |       P        |  |             P  |
714         * ------------------  ------------------  ------------------
715         *
716         */
717 
718        double bsq = distSquared(x00, y00, x11, y11);
719        double csq = distSquared(x00, y00, xpp, ypp);
720        double asq = distSquared(xpp, ypp, x11, y11);
721 
722        if (calculateClockWise) {
723            // c^2 = a^2 + b^2 - 2ab cos C
724            // cos C = (a^2 + b^2 - c^2)/2ab
725            double cosC = (asq + bsq - csq)/(2.* Math.sqrt(bsq)*Math.sqrt(asq));
726            double c = Math.acos(cosC);
727            double theta = c;
728 
729            if (ypp > y00) {
730                // case 0, 1, 2
731                return theta;
732            } else {
733                return 2. * Math.PI - theta;
734            }
735        } else {
736            // a^2 = c^2 + b^2 - 2bc cos A
737            // cos A = (c^2 + b^2 - a^2)/2bc
738            double cosA = (csq + bsq - asq)/(2.* Math.sqrt(bsq)*Math.sqrt(csq));
739            double a = Math.acos(cosA);
740            double theta = a;
741 
742            if (ypp > y00) {
743                // case 0, 1, 2
744                return theta;
745            } else {
746                return 2. * Math.PI - theta;
747            }
748        }
749    }

751    /**
752     * use right triangles when possible to measure
753     *   either the angle between edge 0:1 and 1:point when calculateClockWise=true
754     *   OR the angle between edge 0:1 and 0:point when calculateClockWise=false.
755     * It includes calculations to learn whether the angle is outside or inside the
756     * triangle etc.
757     *
758     * @param x0
759     * @param y0
760     * @param x1
761     * @param y1
762     * @param xPoint
763     * @param yPoint
764     * @param calculateClockWise
765     * @return
766     */
767    private static double angleBetweenPointAndLineWithRightTriangle(float x0,
768                float y0, float x1, float y1, float xPoint, float yPoint, boolean calculateClockWise) {
769 
770            //TODO:  simplify this one day...
771 
772            if (x1 < x0) {
773            throw new IllegalArgumentException("Expecting x0 < x1 or (x0 == x1 and y1 > y1).");
774        }
775 
776        if (calculateClockWise) {
777                // measuring angle between 0:1 and 1:Point
778            float slope01 = (y1 - y0)/(x1 - x0);
779            float slope1p = (yPoint - y1)/(xPoint - x1);
780 
781            if (slope1p == 0) {
782                if (slope01 == 0) {
783                    if (xPoint > x1) {
784                        return Math.PI;
785                    } else {
786                        return 2.*Math.PI;
787                    }
788                }
789                // can use  sine(theta) = yTopPolygon/r and complement or cosine(theta) = xTopPolygon/r
790                double r = Math.sqrt(distSquared(x0, y0, x1, y1));
791                double y =  (y0 - y1);
792 
793                if ((xPoint > x1) && (y0 > y1)) {
794                        double angle = Math.asin(y/r);
795                    return Math.PI - angle;
796                } else if ((xPoint > x1) && (yPoint <= y1) && (x1 > x0)) {
797                        y *= -1;
798                    double angle = Math.asin(y/r);
799                    return Math.PI + angle;
800                } else if ((xPoint > x1) && (yPoint <= y1)) {
801                         y *= -1;
802                     double angle = Math.asin(y/r);
803                     return (2.*Math.PI) - angle;
804 
805                } else if ((xPoint < x1) && (y0 > y1)) {
806                    double angle = Math.asin(y/r);
807                    return 2.*Math.PI - angle;
808 
809                } else if ((xPoint <= x1) && (yPoint >= y1)) {
810                    y *= -1;
811                    double angle = Math.asin(y/r);
812                    return angle;
813                }
814            } else if (slope01 == 0) {
815                // can use cosine(theta) = xTopPolygon/r and complement

1267             * there is no point after the peak, so we create a point right below as a straight line
1268             */

1270 
1271        } else if ((lastIndex == (x.length - 1)) && (yPeakIndex < lastIndex)) {

1291        }
