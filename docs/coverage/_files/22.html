443    public static boolean segmentIsWithinSegment(float x0, float y0, float x1, 
444        float y1, float x2, float y2, float x3, float y3, float generalEps) {

464            y1 = tmp;
465        }

468            x2 = x3;
469            x3 = tmp;

471            y2 = y3;
472            y3 = tmp;
473        }

475

478

487            if (segment01Slope > 0) {
488                if ( (x2 <= x0) && (x1 <= x3) ) {

490                        return true;
491                    }
492                }
493            } else {
494                if ( (x2 <= x0) && (x1 <= x3) ) {
495                    if ( (y0 >= y1) && (y2 >= y3) && (y2 >= y0) && (y1 >= y3) ) {
496                        return true;
497                    }
498                }
499            }

524            }

528        }

530            area *= -1;
531        }

536        for (int i = 0; i < (xPolygon.length - 1); i++) {

539

541        }

545        }

557        if ((yc < 0) && allPointsArePositive) {
558            yc *= -1;
559        }
