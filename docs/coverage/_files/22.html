402    protected void createPoints(int numberOfBackgroundPoints, int[] numberOfClusterPoints,
403        CLUSTER_SEPARATION clusterSeparation,
404        float xmin, float xmax, float ymin, float ymax, SecureRandom sr, boolean useRandomForErrors) {
405 
406        int nClusters = (numberOfClusterPoints == null) ? 0 : numberOfClusterPoints.length;
407 
408        int nTotalPoints = 0;
409 
410        for (int i = 0; i < nClusters; i++) {
411            nTotalPoints += numberOfClusterPoints[i];
412        }
413 
414        nTotalPoints += numberOfBackgroundPoints;
415 
416        // contains points sequentially for: background, cluster[0],...cluster[n-1]
417        x = new float[nTotalPoints];
418        y = new float[nTotalPoints];
419 
420        // compare these to findings.  they are used to generate the clusters
421        xc = new float[nClusters];
422        yc = new float[nClusters];
423

425 
426        /*
427         *  For n = 3
428         *
429         *  |----max/n----|             |             |
430         *  |             |             |             |
431         *         *             *             *
432         *      |  .  |       |  .  |       |  .  |
433         *            |       |                .  |
434         *             d=max/2*n               .  |
435         *                                     r=max/4*n
436         */
437        if (nClusters > 0) {
438 
439            float maxClusterRadius = (xmax - xmin) / (4.0f * nClusters);
440            float minDistanceBetweenClusterCenters;
441            float factor;
442            if (clusterSeparation.ordinal() == CLUSTER_SEPARATION.LARGE.ordinal()) {
443                factor = 4.0f;
444            } else if (clusterSeparation.ordinal() == CLUSTER_SEPARATION.SMALL.ordinal()) {
445                factor = 2.0f;
446            } else {
447                factor = 3.0f;
448            }
449            maxClusterRadius = 0.8f*(xmax - xmin) / (factor * nClusters);
450            minDistanceBetweenClusterCenters = (factor * maxClusterRadius);
451 
452 
453            createRandomSeparatedPoints(sr, xmin + maxClusterRadius, xmax - maxClusterRadius,
454                ymin + maxClusterRadius, ymax - maxClusterRadius,
455                nClusters, xc, yc, minDistanceBetweenClusterCenters);
456 
457            int startOffset = numberOfBackgroundPoints;
458 
459            for (int i = 0; i < nClusters; i++) {
460 
461                int n = numberOfClusterPoints[i];
462 
463                createRandomPointsAroundCenter(sr, maxClusterRadius, n, xc[i], yc[i], x, y, startOffset);
464 
465                startOffset += n;
466            }
467        }
468

473            xErrors[i] = x[i] * 0.03f;
474            yErrors[i] = (float) (Math.sqrt(y[i]));
475        }
